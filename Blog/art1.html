Escrito por Santiago Gallego

Arquitectura del Event Loop

Node.js opera con un modelo de hilo único pero usa un sistema de fases para manejar la concurrencia:

      ┌───────────────────────────┐
      │           Timers          │ ← setTimeout/setInterval
      └─────────────┬─────────────┘
      ┌─────────────┴─────────────┐
      │     Pending Callbacks     │ ← I/O completados
      └─────────────┬─────────────┘
      ┌─────────────┴─────────────┐
      │       Poll Phase          │ ← Nuevos I/O events
      └─────────────┬─────────────┘
      ┌─────────────┴─────────────┐
      │       Check Phase         │ ← setImmediate
      └─────────────┬─────────────┘
      ┌─────────────┴─────────────┐
      │      Close Callbacks      │ ← socket.on('close')
      └───────────────────────────┘

Manejo de errores en Async/Await

Patrón recomendado para evitar "Unhandled Promise Rejections":

// Forma segura
async function fetchData() {
  try {
    const response = await fetch('https://api.example.com');
    const data = await response.json();
    return data;
  } catch (error) {
    console.error('Error en fetchData:', error);
    throw new Error('Fallo al obtener datos'); // Propagación controlada
  }
}

// Uso con Express.js
app.get('/data', async (req, res, next) => {
  try {
    const data = await fetchData();
    res.json(data);
  } catch (err) {
    next(err); // Pasa al middleware de errores
  }
});

Worker Threads para CPU Intensive

Ejemplo de cómo usar hilos para operaciones pesadas:

const { Worker, isMainThread } = require('worker_threads');

if (isMainThread) {
  // Hilo principal
  const worker = new Worker(__filename, { 
    workerData: { n: 40 } // Cálculo de Fibonacci pesado
  });
  
  worker.on('message', (result) => {
    console.log(`Resultado: ${result}`);
  });
} else {
  // Hilo secundario
  const { parentPort, workerData } = require('worker_threads');
  
  function fibonacci(n) {
    return n < 2 ? n : fibonacci(n - 1) + fibonacci(n - 2);
  }
  
  parentPort.postMessage(fibonacci(workerData.n));
}
