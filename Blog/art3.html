Patrones Esenciales para Backend

Los patrones de diseño solucionan problemas comunes de arquitectura. En Python, su implementación es más sencilla gracias a su sintaxis clara.

1. Singleton

Garantiza una única instancia de clase (ideal para conexiones a DB):

class DatabaseConnection:
    _instance = None
    
    def __new__(cls):
        if cls._instance is None:
            cls._instance = super().__new__(cls)
            cls._instance.connect()
        return cls._instance
    
    def connect(self):
        print("Conectando a la base de datos...")

# Uso
db1 = DatabaseConnection()
db2 = DatabaseConnection()
print(db1 is db2)  # True

2. Factory Method

Crea objetos sin especificar la clase exacta:

from abc import ABC, abstractmethod

class Notificador(ABC):
    @abstractmethod
    def enviar(self, mensaje): pass

class EmailNotificador(Notificador):
    def enviar(self, mensaje):
        print(f"Email enviado: {mensaje}")

class SMSNotificador(Notificador):
    def enviar(self, mensaje):
        print(f"SMS enviado: {mensaje}")

def fabrica_notificador(tipo):
    notificadores = {
        'email': EmailNotificador,
        'sms': SMSNotificador
    }
    return notificadores[tipo]()

Dependency Injection en Python

Implementación elegante sin frameworks:

from dataclasses import dataclass

@dataclass
class DatabaseService:
    conn_str: str
    
    def get_users(self):
        print(f"Conectando a {self.conn_str}")
        return ["user1", "user2"]

class UserService:
    def __init__(self, db: DatabaseService):
        self.db = db
    
    def get_all_users(self):
        return self.db.get_users()

# Configuración
dev_db = DatabaseService("server=dev;user=admin")
prod_db = DatabaseService("server=prod;user=admin")

# Uso
user_service = UserService(dev_db)  # Inyectamos dependencia
print(user_service.get_all_users())

Decoradores Avanzados

Implementando Circuit Breaker:

import time
from functools import wraps

class CircuitBreaker:
    def __init__(self, max_fails=3, reset_timeout=10):
        self.max_fails = max_fails
        self.reset_timeout = reset_timeout
        self.fails = 0
        self.last_fail = 0
    
    def __call__(self, func):
        @wraps(func)
        def wrapper(*args, **kwargs):
            if self.fails >= self.max_fails:
                if time.time() - self.last_fail > self.reset_timeout:
                    self.fails = 0  # Reset
                else:
                    raise Exception("Circuit breaker activado")
            
            try:
                result = func(*args, **kwargs)
                self.fails = 0
                return result
            except Exception as e:
                self.fails += 1
                self.last_fail = time.time()
                raise
        
        return wrapper

@CircuitBreaker(max_fails=2, reset_timeout=30)
